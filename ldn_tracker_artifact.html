Connect to Google Sheets

One-time setup syncs your entries to Google Sheets across all devices.

  1. Go to sheets.google.com → create new sheet → name it LDN Tracker Data
  2. Click Extensions → Apps Script
  3. Delete existing code, paste the script below
  4. Click Deploy → New deployment
  5. Type: Web app · Access: Anyone
  6. Click Deploy → copy the URL
  7. Paste URL below and tap Save
Apps Script Code
const SHEET_NAME = "Log"; function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow([ "Date","Dose (mg)","Time Taken", "Energy","Brain Fog","Sleep Quality", "Skin","Gut","Mood","Anxiety", "Overall Wellbeing","Allergy Symptoms", "HRV","Resting HR","Sleep Score", "Cycle Day","Ativan","Propranolol", "Drinks","Notes","Submitted At" ]); } const d = JSON.parse(e.postData.contents); sheet.appendRow([ d.date, d.dose, d.time, d.energy, d.brainfog, d.sleep, d.skin, d.gut, d.mood, d.anxiety, d.overall, d.allergy, d.hrv, d.rhr, d.sleepscore, d.cycle, d.ativan, d.prop, d.drinks, d.notes, new Date().toISOString() ]); return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error",message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return ContentService .createTextOutput(JSON.stringify({rows:[]})) .setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues(); return ContentService .createTextOutput(JSON.stringify({rows:rows})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({rows:[],error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
